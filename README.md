### Installation

  `npm install ip_details`
  
### Usage
```javascript
var ipDetails = require('@pavanvamsi/ip_details');

ipDetails.getIp(function(ipAddress) {
    console.log(ipAddress) // Your IP Address
}

ipDetails.isValidIp('127.0.0.1') // return true or false
```
